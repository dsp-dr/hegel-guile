   ;;; tests/test-integration.scm — end-to-end test (requires hegel-core)
   (when (not (getenv "HEGEL_SERVER_COMMAND"))
     (display "SKIP: HEGEL_SERVER_COMMAND not set\n")
     (exit 0))

   (add-to-load-path (string-append (dirname (current-filename)) "/../src"))
   (use-modules (hegel) (srfi srfi-64))

   (test-begin "integration")

   (define-hegel-test (it-integer-roundtrip tc #:test-cases 50)
     (let* ((n (tc-draw tc (integers #:min-value -1000 #:max-value 1000)))
            (s (number->string n))
            (n2 (string->number s)))
       (unless (= n n2)
         (error "roundtrip failed" n s n2))))

   (let ((failures (run-hegel-tests!)))
     (test-equal "no failures" 0 failures))

   (test-end "integration")
   #+end_src
